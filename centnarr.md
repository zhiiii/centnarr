def identify_scene(business_input):

    prompt = build_prompt_1(business_input)

    response = call_llm(prompt, response_format={"type": "json_object"})

    

    try:

        result = json.loads(response)

    except JSONDecodeError:

        # 兜底：让 AI 重新生成

        result = call_llm(prompt + "\n\n注意：上次输出不是合法 JSON，请严格按 JSON 格式输出")

    

    # 校验必填字段

    required_fields = ["scene", "roles", "pain_points", "emotional_signal"]

    for field in required_fields:

        if field not in result or not result[field]:

            raise ValueError(f"场景识别缺失必填字段: {field}")

    

    return result

    response = call_llm(prompt)

        # 截断到 5 个（取前 5 个）

        result["questions"] = result["questions"][:5]

    

    # 校验：5 个必选维度至少覆盖 3 个

class ConversationState(Enum):

    IDLE = "idle"                      # 空闲

    SCENE_IDENTIFYING = "scene_id"     # 第一轮：场景识别

    ASKING = "asking"                  # 反问中

    ANSWERING = "answering"            # 回答中

    INTEGRATING = "integrating"        # 信息整合中

    CONFIRMING = "confirming"          # 签收中

    PRD_GENERATING = "prd_gen"         # PRD 生成中

    COMPLETED = "completed"            # 完成


class StateMachine:

    def __init__(self, conversation_id):

        self.conversation_id = conversation_id

        self.state = ConversationState.IDLE

        self.round = 0

        self.completion = 0

        self.questions_pending = []

    

    def transition(self, event):

        """状态转移"""

        if self.state == ConversationState.IDLE and event == "first_message":

            self.state = ConversationState.SCENE_IDENTIFYING

            self.round = 1

            

        elif self.state == ConversationState.SCENE_IDENTIFYING and event == "scene_identified":

            self.state = ConversationState.ASKING

            self.questions_pending = [...]  # Prompt 2 生成的问题

            

        elif self.state == ConversationState.ASKING and event == "user_answered":

            self.state = ConversationState.INTEGRATING

            

        elif self.state == ConversationState.INTEGRATING and event == "integrated":

            self.state = ConversationState.ASKING

            self.round += 1

            # 判断是否需要继续

            if self.completion >= 80 and len(self.to_confirm) <= 1:

                self.state = ConversationState.CONFIRMING

                

        elif self.state == ConversationState.CONFIRMING and event == "user_confirmed":

            self.state = ConversationState.PRD_GENERATING

            

        elif self.state == ConversationState.PRD_GENERATING and event == "prd_generated":

            self.state = ConversationState.COMPLETED

# 错误处理：LLM 输出不是合法 JSON

def safe_call_llm(prompt, max_retries=2):

    for i in range(max_retries):

        try:

            response = call_llm(prompt, response_format={"type": "json_object"})

            result = json.loads(response)

            return result

        except JSONDecodeError:

            if i < max_retries - 1:

                # 重试，追加提示

                prompt += "\n\n注意：上次输出不是合法 JSON，请严格按 JSON 格式输出"

            else:

                # 兜底：让用户重新说

                return {

                    "error": "ai_output_invalid",

                    "message": "抱歉，我没理解你的意思，能再说一遍吗？"

                }


# 错误处理：Prompt 3 整合失败

def safe_integrate(previous_doc, new_input, questions):

    try:

        result = call_prompt_3(previous_doc, new_input, questions)

        # 校验必填字段

        validate_integration_result(result)

        return result

    except ValidationError as e:

        # 兜底：保留原文，下次再整理

        return {

            "delta": {"added": [], "modified": [], "confirmed": []},

            "updated_doc": previous_doc,  # 保留上一版

            "user_facing_summary": "我先记下你说的，咱们继续聊。",

            "should_continue": True,
